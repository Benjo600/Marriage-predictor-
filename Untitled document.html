function predictMarriage() {

const inputName = document.getElementById('inputName').value.trim(); const motherTongue = document.getElementById('motherTongue').value;


if (inputName === '') { alert('Please enter your name.'); return;

}


const namesByLanguage = {

'Tamil': ['Karthika', 'Meenakshi', 'Priya', 'Saranya', 'Nithya'],

'Telugu': ['Anusha', 'Lavanya', 'Swathi', 'Divya', 'Anjali'],

'Malayalam': ['Aishwarya', 'Deepa', 'Kavya', 'Sneha', 'Ananya'],

'Hindi': ['Aaradhya', 'Kirti', 'Neha', 'Mitali', 'Preeti']

2023

document.getElementById('wifeInfo').innerText = `Your future wife's name is

${randomWifeName}. ${wifeInfo}`;

document.getElementById('marriageDate').innerText = `In the year ${randomYear}`; document.getElementById('result').style.display = 'block';

}


function getHumorousWifeInfo(motherTongue) { const infoByLanguage = {

};


return infoByLanguage[motherTongue] || 'Get ready for an amazing life with your wonderful future wife!';

}


function shareResult() {
